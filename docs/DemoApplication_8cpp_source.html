185 void DemoApplication::updateCamera() {
186 
187 
188  glMatrixMode(GL_PROJECTION);
189  glLoadIdentity();
190  btScalar rele = m_ele * btScalar(0.01745329251994329547);// rads per deg
191  btScalar razi = m_azi * btScalar(0.01745329251994329547);// rads per deg
192 
193 
194  btQuaternion rot(m_cameraUp,razi);
195 
196 
197  btVector3 eyePos(0,0,0);
198  eyePos[m_forwardAxis] = -m_cameraDistance;
199 
200  btVector3 forward(eyePos[0],eyePos[1],eyePos[2]);
201  if (forward.length2() < SIMD_EPSILON)
202  {
203  forward.setValue(1.f,0.f,0.f);
204  }
205  btVector3 right = m_cameraUp.cross(forward);
206  btQuaternion roll(right,-rele);
207 
208  eyePos = btMatrix3x3(rot) * btMatrix3x3(roll) * eyePos;
209 
210  m_cameraPosition[0] = eyePos.getX();
211  m_cameraPosition[1] = eyePos.getY();
212  m_cameraPosition[2] = eyePos.getZ();
213  m_cameraPosition += m_cameraTargetPosition;
214 
215  if (m_glutScreenWidth == 0 && m_glutScreenHeight == 0)
216  return;
217 
218  btScalar aspect;
219  btVector3 extents;
220 
221  aspect = m_glutScreenWidth / (btScalar)m_glutScreenHeight;
222  extents.setValue(aspect * 1.0f, 1.0f,0);
223 
224 
225  if (m_ortho)
226  {
227  // reset matrix
228  glLoadIdentity();
229 
230 
231  extents *= m_cameraDistance;
232  btVector3 lower = m_cameraTargetPosition - extents;
233  btVector3 upper = m_cameraTargetPosition + extents;
234  //gluOrtho2D(lower.x, upper.x, lower.y, upper.y);
235  glOrtho(lower.getX(), upper.getX(), lower.getY(), upper.getY(),-1000,1000);
236 
237  glMatrixMode(GL_MODELVIEW);
238  glLoadIdentity();
239  //glTranslatef(100,210,0);
240  } else
241  {
242 // glFrustum (-aspect, aspect, -1.0, 1.0, 1.0, 10000.0);
243  glFrustum (-aspect * m_frustumZNear, aspect * m_frustumZNear, -m_frustumZNear, m_frustumZNear, m_frustumZNear, m_frustumZFar);
244  glMatrixMode(GL_MODELVIEW);
245  glLoadIdentity();
246  gluLookAt(m_cameraPosition[0], m_cameraPosition[1], m_cameraPosition[2],
247  m_cameraTargetPosition[0], m_cameraTargetPosition[1], m_cameraTargetPosition[2],
248  m_cameraUp.getX(),m_cameraUp.getY(),m_cameraUp.getZ());
249  }

604 btVector3 DemoApplication::getRayTo(int x,int y)
605 {
606 
607 
608 
609  if (m_ortho)
610  {
611 
612  btScalar aspect;
613  btVector3 extents;
614  aspect = m_glutScreenWidth / (btScalar)m_glutScreenHeight;
615  extents.setValue(aspect * 1.0f, 1.0f,0);
616 
617  extents *= m_cameraDistance;
618  btVector3 lower = m_cameraTargetPosition - extents;
619  btVector3 upper = m_cameraTargetPosition + extents;
620 
621  btScalar u = x / btScalar(m_glutScreenWidth);
622  btScalar v = (m_glutScreenHeight - y) / btScalar(m_glutScreenHeight);
623 
624  btVector3 p(0,0,0);
625  p.setValue((1.0f - u) * lower.getX() + u * upper.getX(),(1.0f - v) * lower.getY() + v * upper.getY(),m_cameraTargetPosition.getZ());
626  return p;
627  }
628 
629  float top = 1.f;
630  float bottom = -1.f;
631  float nearPlane = 1.f;
632  float tanFov = (top-bottom)*0.5f / nearPlane;
633  float fov = btScalar(2.0) * btAtan(tanFov);
634 
635  btVector3 rayFrom = getCameraPosition();
636  btVector3 rayForward = (getCameraTargetPosition()-getCameraPosition());
637  rayForward.normalize();
638  float farPlane = 10000.f;
639  rayForward*= farPlane;
640 
641  btVector3 rightOffset;
642  btVector3 vertical = m_cameraUp;
643 
644  btVector3 hor;
645  hor = rayForward.cross(vertical);
646  hor.normalize();
647  vertical = hor.cross(rayForward);
648  vertical.normalize();
649 
650  float tanfov = tanf(0.5f*fov);
651 
652 
653  hor *= 2.f * farPlane * tanfov;
654  vertical *= 2.f * farPlane * tanfov;
655 
656  btScalar aspect;
657 
658  aspect = m_glutScreenWidth / (btScalar)m_glutScreenHeight;
659 
660  hor*=aspect;
661 
662 
663  btVector3 rayToCenter = rayFrom + rayForward;
664  btVector3 dHor = hor * 1.f/float(m_glutScreenWidth);
665  btVector3 dVert = vertical * 1.f/float(m_glutScreenHeight);
666 
667 
668  btVector3 rayTo = rayToCenter - 0.5f * hor + 0.5f * vertical;
669  rayTo += btScalar(x) * dHor;
670  rayTo -= btScalar(y) * dVert;
671  return rayTo;
672 }
